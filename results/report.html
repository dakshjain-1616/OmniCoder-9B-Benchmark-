        'query_params': query_params

import threading
def parallel_sum(numbers):
    if len(numbers) == 1:
        return numbers[0]
    else:
        mid = len(numbers)//2
        left = threading.Thread(target=parallel_sum, args=(numbers[:mid],))
        right = threading.Thread(target=parallel_sum, args=(numbers[mid:],))
        left.start()
        right.start()
        left.join()
        right.join()
        return left.result + right.result
